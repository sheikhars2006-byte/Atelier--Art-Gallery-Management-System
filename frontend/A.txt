-- READ (SELECT) Operations
-- All Artists
SELECT * FROM Artist;
-- Artworks with Artist Name
SELECT a.Title, a.Category, a.Price, ar.Name AS Artist
FROM Artwork a
JOIN Artist ar ON a.Artist_ID = ar.Artist_ID;
-- Available Artworks
SELECT Title, Price FROM Artwork WHERE Status = 'Available';
-- Sales Details
SELECT s.Sale_ID, a.Title, c.Name, s.Sale_Amount
FROM Sales s
JOIN Artwork a ON s.Artwork_ID = a.Artwork_ID
JOIN Customer c ON s.Customer_ID = c.Customer_ID;
-- Dashboard Stats
SELECT 
 (SELECT COUNT(*) FROM Artist) AS Artists,
 (SELECT COUNT(*) FROM Artwork) AS Artworks,
 (SELECT SUM(Sale_Amount) FROM Sales) AS Revenue;
